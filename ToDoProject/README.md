    localStorage.setItem('tasks', JSON.stringify(this.tasks));
  }

  /**
   * –ú–µ—Ç–æ–¥ –¥–ª—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è –Ω–æ–≤–æ–π –∑–∞–¥–∞—á–∏
   * @param {string} title - –ù–∞–∑–≤–∞–Ω–∏–µ –∑–∞–¥–∞—á–∏
   * @param {string} status - –°—Ç–∞—Ç—É—Å –∑–∞–¥–∞—á–∏ ("–≤ –ø—Ä–æ—Ü–µ—Å—Å–µ" –∏–ª–∏ "–≤—ã–ø–æ–ª–Ω–µ–Ω–æ")
   */
  addTask(title, status) {
    const newTask = {
      id: Date.now().toString(), // –£–Ω–∏–∫–∞–ª—å–Ω—ã–π –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä –∑–∞–¥–∞—á–∏ –Ω–∞ –æ—Å–Ω–æ–≤–µ –≤—Ä–µ–º–µ–Ω–∏
      title,
      status
    };
    this.tasks.push(newTask); // –î–æ–±–∞–≤–ª—è–µ–º –∑–∞–¥–∞—á—É –≤ –º–∞—Å—Å–∏–≤
    this.save(); // –°–æ—Ö—Ä–∞–Ω—è–µ–º –æ–±–Ω–æ–≤–ª—ë–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫
  }

  /**
   * –ú–µ—Ç–æ–¥ –¥–ª—è —É–¥–∞–ª–µ–Ω–∏—è –∑–∞–¥–∞—á–∏ –ø–æ –µ—ë id
   * @param {string} id - –£–Ω–∏–∫–∞–ª—å–Ω—ã–π –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä –∑–∞–¥–∞—á–∏
   */
  deleteTask(id) {
    this.tasks = this.tasks.filter(task => task.id !== id); // –û—Å—Ç–∞–≤–ª—è–µ–º —Ç–æ–ª—å–∫–æ —Ç–µ –∑–∞–¥–∞—á–∏, id –∫–æ—Ç–æ—Ä—ã—Ö –Ω–µ —Å–æ–≤–ø–∞–¥–∞–µ—Ç —Å –ø–µ—Ä–µ–¥–∞–Ω–Ω—ã–º
    this.save(); // –°–æ—Ö—Ä–∞–Ω—è–µ–º –∏–∑–º–µ–Ω–µ–Ω–∏—è
  }

  /**
   * –ú–µ—Ç–æ–¥ –¥–ª—è –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è —Å—Ç–∞—Ç—É—Å–∞ –∑–∞–¥–∞—á–∏
   * –ï—Å–ª–∏ –∑–∞–¥–∞—á–∞ "–≤ –ø—Ä–æ—Ü–µ—Å—Å–µ", —Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è "–≤—ã–ø–æ–ª–Ω–µ–Ω–æ", –∏ –Ω–∞–æ–±–æ—Ä–æ—Ç
   * @param {string} id - –£–Ω–∏–∫–∞–ª—å–Ω—ã–π –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä –∑–∞–¥–∞—á–∏
   */
  toggleStatus(id) {
    const task = this.tasks.find(t => t.id === id); // –ò—â–µ–º –Ω—É–∂–Ω—É—é –∑–∞–¥–∞—á—É
    if (task) {
      task.status = task.status === '–≤ –ø—Ä–æ—Ü–µ—Å—Å–µ' ? '–≤—ã–ø–æ–ª–Ω–µ–Ω–æ' : '–≤ –ø—Ä–æ—Ü–µ—Å—Å–µ'; // –ü–µ—Ä–µ–∫–ª—é—á–∞–µ–º —Å—Ç–∞—Ç—É—Å
      this.save(); // –°–æ—Ö—Ä–∞–Ω—è–µ–º
    }
  }

  /**
   * –ú–µ—Ç–æ–¥ –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –∑–∞–¥–∞—á –ø–æ —Ñ–∏–ª—å—Ç—Ä–∞–º –ø–æ–∏—Å–∫–∞ –∏ —Å—Ç–∞—Ç—É—Å–∞
   * @param {string} search - –ü–æ–∏—Å–∫–æ–≤–∞—è —Å—Ç—Ä–æ–∫–∞
   * @param {string} status - –§–∏–ª—å—Ç—Ä –ø–æ —Å—Ç–∞—Ç—É—Å—É ("–≤—Å–µ", "–≤ –ø—Ä–æ—Ü–µ—Å—Å–µ", "–≤—ã–ø–æ–ª–Ω–µ–Ω–æ")
   * @returns {Array} - –û—Ç—Ñ–∏–ª—å—Ç—Ä–æ–≤–∞–Ω–Ω—ã–π –º–∞—Å—Å–∏–≤ –∑–∞–¥–∞—á
   */
  getFilteredTasks(search, status) {
    return this.tasks.filter(task => {
      const matchSearch = task.title.toLowerCase().includes(search); // –°—Ä–∞–≤–Ω–∏–≤–∞–µ–º —Å –ø–æ–∏—Å–∫–æ–≤–æ–π —Å—Ç—Ä–æ–∫–æ–π
      const matchStatus = status === '–≤—Å–µ' || task.status === status; // –ü—Ä–æ–≤–µ—Ä—è–µ–º, —Å–æ–≤–ø–∞–¥–∞–µ—Ç –ª–∏ —Å—Ç–∞—Ç—É—Å
      return matchSearch && matchStatus; // –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Ç–æ–ª—å–∫–æ —Ç–µ –∑–∞–¥–∞—á–∏, –∫–æ—Ç–æ—Ä—ã–µ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—Ç –æ–±–æ–∏–º —É—Å–ª–æ–≤–∏—è–º
    });
  }
}
